package own;

public class MaxNumberFromArray {

	public static void main(String[] args) {
		int [] arr= {1,2,3,23,123};
		int max=0;
		
	for (int i= 0; i<arr.length; i++) {
		if (arr[i]>max) {
			max=arr[i];
		}
	}
	System.out.println(max);
	}
}
